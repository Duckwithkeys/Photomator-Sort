        private static func escape(_ value: String) -> String {
            var result = ""
            result.reserveCapacity(value.count + value.count / 10)
            for char in value {
                switch char {
                case "&":  result += "&amp;"
                case "<":  result += "&lt;"
                case ">":  result += "&gt;"
                case "\"": result += "&quot;"
                default:   result.append(char)
                }
            }
            return result
        }

        private static func unescape(_ value: String) -> String {
            var result = value
            // Order matters: unescape &amp; last to avoid double-unescaping
            result = result.replacingOccurrences(of: "&lt;",   with: "<")
            result = result.replacingOccurrences(of: "&gt;",   with: ">")
            result = result.replacingOccurrences(of: "&quot;", with: "\"")
            result = result.replacingOccurrences(of: "&apos;", with: "'")
            result = result.replacingOccurrences(of: "&amp;",  with: "&")
            return result
        }

    Alternatively, write unescape() as a single-pass scanner too (similar
    approach to escape, detecting & and reading ahead to the semicolon).

4J. CHUNK FILE SIZE COMPUTATION IN FileTransferService
    File: FileTransferService.swift, lines 86-99

    The totalBytes pre-computation spawns one Task per file just to call
    FileManager.attributesOfItem (a single stat() syscall). For 1,500 files that
    is 1,500 task objects for trivially fast work. Replace with chunked parallelism:

        let chunkSize = max(50, files.count / ProcessInfo.processInfo.activeProcessorCount)
        let totalBytes = await withTaskGroup(of: Int64.self) { group in
            for chunkStart in stride(from: 0, to: files.count, by: chunkSize) {
                let slice = Array(files[chunkStart..<min(chunkStart + chunkSize, files.count)])
                group.addTask {
                    let fm = FileManager.default
                    return slice.reduce(Int64(0)) { sum, url in
                        let size = (try? fm.attributesOfItem(atPath: url.path)[.size] as? Int64) ?? 0
                        return sum + size
                    }
                }
            }
            var total: Int64 = 0
            for await partial in group { total += partial }
            return total
        }

    Also store the per-file size from this pre-computation so executeJob() does not
    need to stat() each file a second time when tracking bytesPerSecond progress.

    Same fix applies to RoutedTransferService.swift:71.
